# 最小生成树模板：
```c++
#include <cstdio>
#include <cmath>
#include <cstdlib>
#include <cstring>
const int N  =110;
const int INF = 0x3f3f3f3f;
int m[N][N];
int lowcost[N];
bool add[N];
int Prim(int s, int n)
{
    int sumweight = 0;
    int i, j, k = 0;
    memset(lowcost, 0, sizeof(lowcost));//初始化最小花费
    for(i = 1; i <= n; i++)
    {
        lowcost[i] = m[s][i];//初始化与起点相连的点之间的权值
        add[i] = false;//将所有点都初始化为未访问过
    }
    add[s] = true;//起点加入访问过的点集中
    for(i = 1; i < n; i++)
    {
        int minl = INF;
        int v = -1;
        for(j = 1; j <= n; j++)
        {
            if(!add[j] && lowcost[j] < minl)//寻找未访问的点中与起点距离最近的点
            {
                minl = lowcost[j];
                v = j;
            }
        }
        if(v != -1)
        {
            add[v] = true;
            sumweight += lowcost[v];
            for(j = 1; j <= n; j++)
            {
                if(!add[j] && m[v][j] < lowcost[j])//松弛，逐步构建最小生成树
                {
                    lowcost[j] = m[v][j];
                }
            }
        }
    }
    return sumweight;
}
int main()
{
    int b, d, n;
    char a, c, t;
    while(~scanf("%d", &n) && n)
    {
        getchar();
        memset(m, INF, sizeof(m));//构建最小生成树，所以将图中点与点间的所有距离(权值)初始化最大
        for(int i = 1; i < n; i++)
        {
            scanf("%c %d", &a, &b);
            getchar();
            while(b--)
            {
                scanf("%c %d", &c, &d);
                m[(int)(a-'A'+1)][(int)(c-'A'+1)] = d;
                m[(int)(c-'A'+1)][(int)(a-'A'+1)] = d;
                getchar();
            }
        }
        printf("%d\n", Prim(1, n));
    }
    return 0;
}
```
