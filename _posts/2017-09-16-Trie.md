# Trie树模板
```c++
#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
const int N = 26;
typedef struct Trie
{
    Trie *next[N];
    int v;//根据需要变化
}Trie;
Trie root;
void CreateTrie(char *str)
{
    int len = (int)strlen(str);
    Trie *p = &root, *q;
    for(int i = 0; i < len; i++)
    {
        int id = str[i] - 'a';
        if(p->next[id] == NULL)
        {
            q = (Trie *)malloc(sizeof(root));
            q->v = 1;//初始化为1
            for(int j = 0; j < N; j++)
            {
                q->next[j] = NULL;
            }
            p->next[id] = q;
            p = p->next[id];
        }
        else
        {
            p->next[id]->v++;
            p = p->next[id];
        }
    }
    //p->v = -1;//字符串的最后一个字母对应的v为-1
}
int FindTrie(char *str)
{
    int len = (int)strlen(str);
    Trie *p = &root;
    for(int i = 0; i < len; i++)
    {
        int id = str[i] - 'a';
        p = p->next[id];
        if(p == NULL)//不存在以str为前缀的字符串
        {
            return 0;
        }
        /*
        if(p->v == -1)//str是字符串集合中某个字符串的前缀
        {
            return -1;
        }
        */
    }
    //return -1;//str是字符串集合中某个字符串的前缀
    return p->v;
}

int main()
{
    char str[N];
    for(int i = 0; i < N; i++)
    {
        root.next[i] = NULL;
    }
    while(gets(str) && str[0] != '\0')
    {
        CreateTrie(str);
    }
    memset(str, 0, sizeof(str));
    while(~scanf("%s", str))
    {
        printf("%d\n", FindTrie(str));
    }
    return 0;
```
